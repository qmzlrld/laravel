<IfModule mod_rewrite.c>
    RewriteEngine On

    <FilesMatch "^(?:\.env|composer\.(json|lock)|artisan|package\.json|package-lock\.json|npm-debug\.log)$">
        Require all denied
    </FilesMatch>

    # If request already targets /public, serve normally
    RewriteRule ^public/ - [L]

    # If file/dir exists under public, serve it from public
    RewriteCond %{DOCUMENT_ROOT}/public/%{REQUEST_URI} -f [OR]
    RewriteCond %{DOCUMENT_ROOT}/public/%{REQUEST_URI} -d
    RewriteRule ^(.*)$ public/$1 [L]

    # Otherwise internally rewrite everything to public/index.php
    RewriteRule ^(.*)$ public/index.php [L]
</IfModule>

# Prefer index from public
DirectoryIndex public/index.php index.php index.html
